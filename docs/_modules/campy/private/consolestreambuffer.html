[docs]    def print(self, *objects, sep='', end='\n', flush=False, stderr=False):
        buffer = _io.StringIO()
        __builtins__.print(*objects, sep=sep, end=end, file=buffer, flush=True)
        data = buffer.getvalue()
        _platform.Platform().put_console(data, stderr=stderr)
